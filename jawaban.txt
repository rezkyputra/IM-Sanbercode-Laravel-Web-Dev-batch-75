1. Buat Database
CREATE DATABASE library;

2. Buat Table
- Genres
CREATE TABLE genres( id int AUTO_INCREMENT PRIMARY KEY, name varchar(255) NOT null );

- Books
CREATE TABLE books( id int AUTO_INCREMENT PRIMARY KEY, title varchar(255) NOT null, price int NOT null, stock int NOT null, genre_id int NOT null, FOREIGN KEY(genre_id) REFERENCES genres(id) );
